# API 服务部署

# 1. 安装项目依赖

npm install

# 2. 编写配置文件

// 开发环境配置文件 config/config.local.js
// 生产环境配置文件 config/config.prod.js

{
  keys: '随机安全 key',

  // MongoDB configs.
  mongodb: {
    host: '127.0.0.1',
    port: 27017,
    database: '数据库名称',
    user: '数据库账号',
    pass: '数据库密码',
  },
}

# 3. 运行 API 服务

npm run start # 启动运行 API 服务
npm run stop # 停止运行 API 服务

# 4. 执行初始化数据库脚本

# 仅在第一次配置时执行， 默认创建超级管理员ROOT
node scripts/init-admin.js  管理员用户名 管理员密码

# Web 客户端部署

npm run web:build
