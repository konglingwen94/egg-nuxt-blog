# 添加标签

POST /admin/tags

# Request

Field Type Description Required
name String 标签名称 Yes
cover File 封面 Yes

# Response

HTTP/1.1 201 Created

{
  "name": "Nuxt.js",
  "id": "5d9d9ce3246364202c897068",

# 编辑标签

PATCH /admin/tags/:id

# Request

Field Type Description Required
name String 标签名称 Yes
cover File 标签封面 Yes

# Response

HTTP/1.1 204 No Content

# 删除标签

DELETE /admin/tags/:id

# Response

HTTP/1.1 204 No Content
